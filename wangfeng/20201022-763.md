  class Solution {
      public List<Integer> partitionLabels(String S) {

          List<Integer> list = new ArrayList<Integer>();
          int slow = 0;
          int cur = slow;
          int n = S.length();
          int cur0 = 0;
          while(slow < n){

              int fast = cur + 1;
              while(fast < n){
                  if(S.charAt(fast) == S.charAt(slow)){
                      cur = fast;
                  }
                  fast += 1;
              }       
              if(slow == cur){

                  list.add(cur + 1 - cur0);
                  cur0 = cur + 1 ;

              }
              slow += 1;
          }

          return list;
      }
  }
